void automation.SendAnnualPlanpprovedEmail(Int accID)
{
accObj = zoho.crm.getRecordById("Accounts",accID);
if(accObj != null && accObj.isEmpty() == false)
{
	distributorName = ifnull(accObj.get("Account_Name"),"Distributor");
	distributorEmail = ifnull(accObj.get("Account_Email"),"");
	if(distributorEmail == "")
	{
		return;
	}
	mailSubject = "Annual Plan Approved";
	htmlMessage = "<html><body style='font-family:Arial,sans-serif;background-color:#f3f4f6;padding:20px;'><div style='max-width:600px;margin:auto;background:#ffffff;border-radius:12px;padding:24px;box-shadow:0 8px 24px rgba(0,0,0,0.08);'><h2 style='color:#16a34a;text-align:center;margin-top:0;'>Annual Plan Approved</h2><p style='font-size:14px;color:#111827;'>Dear <b>" + distributorName + "</b>,</p><p style='font-size:14px;color:#374151;'>We are pleased to inform you that your submitted <b>Annual Plan</b> has been <b>approved</b>.</p><div style='background:#ecfdf5;border-left:4px solid #16a34a;padding:14px;margin:16px 0;border-radius:6px;'><p style='margin:0;font-size:13px;color:#065f46;'><b>Status:</b> Approved</p></div><p style='font-size:14px;color:#374151;'>You may now proceed with further planning and execution as per the approved Annual Plan.</p><hr style='border:none;border-top:1px solid #e5e7eb;margin:20px 0;'><p style='font-size:12px;color:#6b7280;text-align:center;margin-bottom:0;'>This is an automated email. Please do not reply.</p></div></body></html>";
	sendmail
	[
		from :zoho.loginuserid
		to :distributorEmail
		subject :mailSubject
		message :htmlMessage
	]
}
}
